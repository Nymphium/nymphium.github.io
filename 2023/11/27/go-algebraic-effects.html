4.4. symmetric coroutineсЂ«т«ЪУБЁ
package main

type T struct {
    ch chan any
}

func New(f func(any) any) *T {
    ch := make(chan any)
    co := &T{ch}

    go func() {
        var res any
        defer func() {
            ch <- res
        }()

        res = f(<-ch)
    }()

    return co
}

func (co *T) Transfer(v any) any {
    co.ch <- v
    return <-co.ch
}

func main() {
    var f1, f2, f3 *T

    f1 = New(func(any) any {
        fmt.Println("f1")
        return f2.Transfer(0)
    })

    f2 = New(func(any) any {
        fmt.Println("f2")
        return f3.Transfer(0)
    })

    f3 = New(func(any) any {
        fmt.Println("f3")
        return 0
    })

    f1.Transfer(0)
}
// f1
// f2
// f3

УАе6.1сЂФсЂѓсѓІсѓѕсЂєсЂФ№йцasymmetric coroutinesсЂ»тЉ╝сЂ│тЄ║сЂЌсЂФУдфтГљжќбС┐ѓсЂїсЂѓсѓІ№йА resumeсЂїтЉ╝сЂ│тЄ║сЂЎcoroutineсѓњТїЄт«џсЂДсЂЇсѓІсЂ«сЂФт»ЙсЂЌсЂд№йцyieldсЂ»сЃфсѓ┐сЃ╝сЃ│сЃЮсѓцсЃ│сЃѕсѓњТїЄт«џсЂДсЂЇсЂџ№йцтЉ╝сЂ│тЄ║сЂЌтЁЃсЂ«УдфсЂФТѕ╗сѓІ№йА сЂцсЂЙсѓі(сЂЊсЂЊсЂДУФќуљєсЂ«жБЏУ║Ї)сѓ╣сѓ┐сЃЃсѓ»сЂїСй┐сЂѕсЂЙсЂЎсЂГ(тЏ│5.1)!

УЅ»сЂётЏ│сЂфсЂ«сЂДGopherсЂЈсѓЊсѓѓТёЪт┐ЃсЂЌсЂдсЂЙсЂЎ№йА ChannelsсЂ»send/recvсѓњСИАТќ╣тЁ╝сЂГсЂдсЂёсѓІсЂ«сЂД№йцsendсЂЎсѓІсЂесЂЇсЂФchannelсѓњstackсЂФуфЂсЂБУЙ╝сѓЊсЂДrecvсЂЌсЂцсЂцgoroutineсѓњжЂиуД╗сЂЌ№йцтЉ╝сЂ░сѓїсЂЪтЂ┤сЂїpopсЂЌсЂдsendсЂЌсЂдrecvтЂ┤сЂФТѕ╗сѓІ№йА жќбТЋ░тЉ╝сЂ│тЄ║сЂЌсЂ«сѓ│сЃ╝сЃФсѓ╣сѓ┐сЃЃсѓ»сЂ«сѓѕсЂєсЂфсѓѓсЂ«сѓњ№йцchannelсЂ«сѓ╣сѓ┐сЃЃсѓ»сЂДУАеуЈЙсЂЎсѓІсЂЊсЂесЂФсЂфсѓІ№йА жЏБсЂЌсЂёсЂЊсЂесѓњУђЃсЂѕсЂџсЂФсЂЊсЂ«call stackсЂ»сѓ░сЃГсЃ╝сЃљсЃФсЂФ1сЂцТїЂсЂцсѓѕсЂєсЂФсЂЌсѓѕсЂє(5.2)№йА

5.2. asymmetric stackful coroutineсЂ«т«ЪУБЁ
package main

type stack []*T

func (s *stack) push(t *T) {
    *s = append(*s, t)
}
func (s *stack) pop() *T {
    t := (*s)[len(*s)-1]
    *s = (*s)[:len(*s)-1]
    return t
}

// global stack
var s stack = make([]*T, 0)

type T struct {
    ch chan any
}

// сЂЊсѓїсЂ»тљїсЂў
func New(f func(any) any) *T {
    ch := make(chan any)
    co := &T{ch}
    go func() {
        var res any
        defer func() {
            ch <- res
        }()
        res = f(<-ch)
    }()
    return co
}

// channelсѓњpushсЂЌсЂдтђцсѓњжђЂсЂБсЂдУ┐ћсЂБсЂдсЂЈсѓІсЂ«сѓњтЙЁсЂц
func (co *T) Resume(v any) any {
    s.push(co)
    co.ch <- v
    return <-co.ch
}

// popсЂЌсЂдтђцсѓњжђЂсѓіУ┐ћсЂЌсЂдсЂЋсѓЅсЂФтђцсЂїТЮЦсѓІсЂ«сѓњтЙЁсЂц
func Yield(v any) any {
    ch := s.pop().ch
    ch <- v
    return <-ch
}

func main() {
    var f1, f2 *T
    f1 = New(func(mark any) any {
        name := f2.Resume(0)
        fmt.Printf("hello, %s%s\n", name, mark)
        return 0
    })
    f2 = New(func(any) any {
        Yield("world")
        return 0
    })

    f1.Resume("!")
}

// hello, world!

6. StacklessсЂ»сѓ«сЃќ

сЂЋсЂд№йцstackless asymmetric coroutinesсЂ«УЕ▒сЂДуиасѓЂсѓІ№йА Тћ╣сѓЂсЂд№йцstackless coroutinesсЂ»GoсЂДсЂ»сѓесЃЪсЃЦсЃгсЃ╝сЃѕсЂДсЂЇсЂфсЂё№йА сЂФсЂЙсЂесѓЂсЂдсЂѓсѓІсЂесЂісѓі№йцstackless coroutinesсЂ»жќбТЋ░тЉ╝сЂ│тЄ║сЂЌсѓњсЂЙсЂЪсЂљсЂЊсЂесЂїсЂДсЂЇсЂфсЂё№йА

Stackless asymmetric coroutinesсЂ«ТюгУ│фсЂ»№йцжќбТЋ░тЉ╝сЂ│тЄ║сЂЌсѓњсЂЙсЂЪсЂљсЂЊсЂесЂїсЂДсЂЇсЂфсЂё№йцсЂесЂёсЂєтѕХу┤ёсЂФсЂѓсѓІ№йА "coroutineсЂїжќбТЋ░тЉ╝сЂ│тЄ║сЂЌсѓњсЂЙсЂЪсЂљ"сЂесЂ»№йцresumeсЂЌсЂЪcoroutineсЂ«СИГсЂДжќбТЋ░сѓњтЉ╝сЂ│тЄ║сЂЌ№йцсЂЮсЂ«СИГсЂДyieldсЂЎсѓІ№йцсЂесЂёсЂєсѓ▒сЃ╝сѓ╣сЂДсЂѓсѓІ№йА сЂЊсѓїсЂї"сЂДсЂЇсЂфсЂё"сЂесЂ»№йцСЙІсЂѕсЂ░JSсЂ«GeneratorsсЂ«сѓѕсЂєсЂФfunction*сЂ«СИГсЂДсЂЌсЂІyieldсЂїСй┐сЂѕсЂфсЂёсЂесЂёсЂєтѕХу┤ёсЂДУАеуЈЙсЂЋсѓїсѓІ№йА

GoсЂДсЂ»№йцсЂЊсЂ«сѓѕсЂєсЂфжќбТЋ░тЉ╝сЂ│тЄ║сЂЌсЂФтѕХжЎљсѓњУеГсЂЉсѓІТќ╣Т│ЋсЂїсЂфсЂёсЂЪсѓЂ№йцstacklessnessсѓњУАеуЈЙсЂЎсѓІсЂЊсЂесЂ»С╗ісЂ«сЂесЂЊсѓЇСИЇтЈ»УЃйсЂДсЂѓсѓІ№йА сЂЌсЂІсЂЌ№йцсЂЊсЂ«stacklessnessсЂ»№йцcoroutinesсЂ«call stackсѓњжќбТЋ░сѓњсЂЙсЂЪсЂёсЂДу«АуљєсЂЌсЂфсЂёсЂЊсЂесЂФсѓѕсѓІсЃЉсЃЋсѓЕсЃ╝сЃъсЃ│сѓ╣тљЉСИісЂесЂёсЂєсЃЄсѓХсѓцсЃ│сЃЂсЃДсѓцсѓ╣сѓё№йцcoroutineсЂћсЂесЂФжќбТЋ░сЂ«сѓ│сЃ╝сЃФсѓ╣сѓ┐сЃЃсѓ»сѓњу«АуљєсЂДсЂЇсЂфсЂёсЂесЂёсЂєсЃЕсЃ│сѓ┐сѓцсЃасЂ«тѕХу┤ёсѓњтЈЇТўасЂЌсЂдсЂёсѓІ(тЈ»УЃйТђДсЂїсЂѓсѓІ)№йА сЂЌсЂЪсЂїсЂБсЂд№йцGoсѓњтѕЮсѓЂсЂесЂЌсЂЪуЅ╣т«џсЂ«тйбсЂ«жќбТЋ░тЉ╝сЂ│тЄ║сЂЌсѓњУАеуЈЙсЂЎсѓІТЕЪУЃйсЂ«уёАсЂёУеђУфъсЂДstackless coroutinesсѓњт«ЪУБЁсЂДсЂЇсЂфсЂёсЂЊсЂесѓњ№йцсЂЮсЂЊсЂЙсЂДТѓ▓Уд│сЂЎсѓІт┐ЁУдЂсЂ»сЂфсЂё4№йА
